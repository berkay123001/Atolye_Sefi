# Fibonacci fonksiyonu
def fibonacci(n):
    if n <= 1:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

# İlk 15 Fibonacci sayısını hesapla
print("📊 İlk 15 Fibonacci Sayısı:")
for i in range(15):
    result = fibonacci(i)
    print(f"fibonacci({i}) = {result}")

# Performans testi
import time
print("\n⏱️ Performans Testi:")
start_time = time.time()
result = fibonacci(30)
end_time = time.time()
print(f"fibonacci(30) = {result}")
print(f"Süre: {end_time - start_time:.4f} saniye")

# Sonuçları dosyaya kaydet
with open("fibonacci_results.txt", "w") as f:
    f.write("Fibonacci Sonuçları\n")
    f.write("==================\n")
    for i in range(15):
        f.write(f"fibonacci({i}) = {fibonacci(i)}\n")
    f.write(f"\nPerformans: fibonacci(30) = {fibonacci(30)}\n")

print("\n✅ Sonuçlar fibonacci_results.txt dosyasına kaydedildi!")
